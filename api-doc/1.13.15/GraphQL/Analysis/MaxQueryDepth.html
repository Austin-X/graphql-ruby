def initialize(max_depth)
  disallow_excessive_depth = ->(query, depth) {
    if depth > max_depth
      GraphQL::AnalysisError.new("Query has depth of #{depth}, which exceeds max depth of #{max_depth}")
    else
      nil
    end
  }
  super(&disallow_excessive_depth)
end
